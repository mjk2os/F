Aim: Applying Conditional Filters using subset() or filter() in R

üîπ Step 1: Create or Load a Dataset
students <- data.frame(
  Name = c("Asha", "Ravi", "Neha", "Kiran", "Meena", "Raj"),
  Age = c(20, 21, 19, 22, 20, 23),
  Marks = c(85, 90, 88, 76, 92, 67),
  Gender = c("F", "M", "F", "M", "F", "M")
)

üîπ Step 2: Apply Filters using subset() (Base R)

high_marks <- subset(students, Marks > 80)
print(high_marks)
female_students <- subset(students, Gender == "F")

filtered_data <- subset(students, Age > 20 & Marks > 80)

üîπ Step 3: Apply Filters using filter() (from dplyr)

print(male_low_marks)

üîπ Step 4: Combining Conditions

print(age_range)
filter(students, Gender == "F" | Marks > 90)

Practical No. 10

Aim: Creating new variables using transformations and calculations

students <- data.frame(
  Name = c("Asha", "Ravi", "Neha", "Kiran", "Meena", "Raj"),
  Marks1 = c(85, 90, 88, 76, 92, 67),
  Marks2 = c(78, 95, 82, 80, 89, 70)
)
students$Total <- students$Marks1 + students$Marks2

students$Average <- students$Total / 2
students$Percent <- (students$Total / 200) * 100

students$Grade <- ifelse(students$Percent >= 75, "Distinction",
                  ifelse(students$Percent >= 60, "First Class",
                  ifelse(students$Percent >= 50, "Second Class", "Fail")))
print(students)

  Name = c("Asha", "Ravi", "Neha"),
  Marks_Math = c(85, 90, 88),
  Marks_Science = c(78, 95, 82)
)

long_data <- pivot_longer(
  students,
  cols = c(Marks_Math, Marks_Science),
  names_to = "Subject",
  values_to = "Marks"
)

wide_data <- pivot_wider(
  long_data,

)
